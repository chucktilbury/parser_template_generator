grammar
    : non_terminal_rule  SEMI
    | grammar non_terminal_rule SEMI
    ;

non_terminal_rule
    : NON_TERMINAL COLON rule_element_list
    ;

rule_element_list
    : rule_element
    | rule_element_list rule_element
    ;

rule_element
    : terminal_rule_element
    | non_terminal_rule_element
    ;

terminal_rule_element
    : NON_TERMINAL
    | TERMINAL_NAME
    | TERMINAL_OPER
    | TERMINAL_SYMBOL
    ;

non_terminal_rule_element
    : or_function
    | zero_or_more_function
    | zero_or_one_function
    | one_or_more_function
    | grouping_function
    ;

or_function
    : rule_element PIPE
    ;

zero_or_more_function
    : rule_element QUESTION
    ;

zero_or_one_function
    : rule_element STAR
    ;

one_or_more_function
    : rule_element PLUS
    ;

grouping_function
    : OPAREN rule_element_list CPAREN
    ;

grammar
non_terminal_rule
rule_element_list
rule_element
terminal_rule_element
non_terminal_rule_element
or_function
zero_or_more_function
zero_or_one_function
one_or_more_function
grouping_function
